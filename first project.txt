#include <stdio.h>

int main()
{
    int sub1,sub2,sub3,sub4,sub5,perc;
    printf("Hello Student \n");
    printf("Get your grade point here \n");
    printf("Enter your subject 1 score: \n");
    scanf("\n %d",&sub1);
    printf("Enter your subject 2 score: \n");
    scanf("\n %d",&sub2);
    printf("Enter your subject 3 score: \n");
    scanf("\n %d",&sub3);
    printf("Enter your subject 4 score: \n");
    scanf("\n %d",&sub4);
    printf("Enter your subject 5 score: \n");
    scanf("\n %d",&sub5);
    perc=(sub1+sub2+sub3+sub4+sub5)/5;
    printf("Your percentage is %d \n",perc);
    if(perc<50){
        printf("Don't worry!! Marks don't determine your future!! \n");
        printf("Do well next time. \n ");
    }
    else if (50<=perc&&perc<60){
        printf("Your grade point is 6.00 \n");
    }
    else if (60<=perc&&perc<70){
        printf("Your grade point is 7.00 \n");
    }
    else if (70<=perc&&perc<80){
        printf("Your grade point is 8.00 \n");
    }
    else if (80<=perc&&perc<90){
        printf("Your grade point is 9.00 \n");
    }
    else if (90<=perc&&perc<100){
        printf("Your grade point is 10.00 \n");
        printf("Your are brilliant and will do well in life!! \n");
    }
    return 0;
}
